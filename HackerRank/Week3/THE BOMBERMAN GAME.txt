char** explode(int rows, int cols, char** grid){
    bool** cleared = malloc(rows * sizeof(bool*));
    for(int i = 0; i < rows; i++){
        cleared[i] = calloc(cols, sizeof(bool));
    }
    for(int i = 0; i < rows; i++){
        for(int j = 0; j < cols; j++){
            if(grid[i][j] == 'O'){
                cleared[i][j] = true;
                if(i > 0)cleared[i - 1][j] = true;
                if(i < rows-1)cleared[i + 1][j] = true;
                if(j >  0)cleared[i][j-1] = true;
                if(j < cols-1)cleared[i][j+1] = true;
            }
        }
    }
    char** newGrid = malloc(rows * sizeof(char*));
    for(int i = 0; i < rows; i++){
        newGrid[i] = malloc((cols+1) * sizeof(char));
        for(int j = 0; j < cols; j++){
            newGrid[i][j] = cleared[i][j] ? '.' : 'O';
        }
        newGrid[i][cols] = '\0';
    }
    for(int i = 0; i < rows; i++) free(cleared[i]);
    free(cleared);
    return newGrid;
}
char** fullGrid(int rows, int cols){
    char** g = malloc(rows * sizeof(char*));
    for(int i = 0; i < rows; i++){
        g[i] = malloc((cols+1) * sizeof(char));
        for(int j = 0; j < cols; j++)
            g[i][j] = 'O';
        g[i][cols] = '\0';
    }
    return g;
}

char** bomberMan(int n, int grid_count, char** grid, int* result_count) {
    int rows = grid_count;
    int cols = strlen(grid[0]);
    *result_count = rows;
    if(n == 1){
        char** res = malloc(rows * sizeof(char*));
        for(int i = 0; i < rows; i++){
            res[i] = malloc((cols+1) * sizeof(char));
            strcpy(res[i], grid[i]);
        }
        return res;
    }
    if(n % 2 == 0){
        return fullGrid(rows, cols);
    }
    char** first = explode(rows, cols, grid);
    if(n % 4 == 3){
        return first;
    }else{
        char** second = explode(rows,cols,first);
        for(int i = 0; i < rows; i++) free(first[i]);
        free(first);
        return second;
    }

}

char* readline() {
    size_t alloc_length = 1024;
    size_t data_length = 0;

    char* data = malloc(alloc_length);

    while (true) {
        char* cursor = data + data_length;
        char* line = fgets(cursor, alloc_length - data_length, stdin);

        if (!line) {
            break;
        }

        data_length += strlen(cursor);

        if (data_length < alloc_length - 1 || data[data_length - 1] == '\n') {
            break;
        }

        alloc_length <<= 1;

        data = realloc(data, alloc_length);

        if (!data) {
            data = '\0';

            break;
        }
    }

    if (data[data_length - 1] == '\n') {
        data[data_length - 1] = '\0';

        data = realloc(data, data_length);

        if (!data) {
            data = '\0';
        }
    } else {
        data = realloc(data, data_length + 1);

        if (!data) {
            data = '\0';
        } else {
            data[data_length] = '\0';
        }
    }

    return data;
}

char* ltrim(char* str) {
    if (!str) {
        return '\0';
    }

    if (!*str) {
        return str;
    }

    while (*str != '\0' && isspace(*str)) {
        str++;
    }

    return str;
}

char* rtrim(char* str) {
    if (!str) {
        return '\0';
    }

    if (!*str) {
        return str;
    }

    char* end = str + strlen(str) - 1;

    while (end >= str && isspace(*end)) {
        end--;
    }

    *(end + 1) = '\0';

    return str;
}

char** split_string(char* str) {
    char** splits = NULL;
    char* token = strtok(str, " ");

    int spaces = 0;

    while (token) {
        splits = realloc(splits, sizeof(char*) * ++spaces);

        if (!splits) {
            return splits;
        }

        splits[spaces - 1] = token;

        token = strtok(NULL, " ");
    }

    return splits;
}

int parse_int(char* str) {
    char* endptr;
    int value = strtol(str, &endptr, 10);

    if (endptr == str || *endptr != '\0') {
        exit(EXIT_FAILURE);
    }

    return value;
}
